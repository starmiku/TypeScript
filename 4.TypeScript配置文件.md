# TypeScript配置文件

tsconfig.json是TypeScript项目的配置文件，用于指定编译选项和项目设置。

## 基础tsconfig.json配置

最基础的tsconfig.json文件

```ts
{
    "compilerOptions": {
        "target": "ES2020",//编译目标JavaScript版本
        "module": "commonjs",//使用的模块系统
        "strict": true,//启用所有严格类型检查
        "outDir": "./dist"//输出目录
    },
    "include": ["src/**/*"],
    "exclude": ["node_modules", "dist"]
}
```

### 编译目标版本

使用target指定编译到的JavaScript版本。

```ts
// tsconfig.json
{
    "compilerOptions": {
        // ES3, ES5, ES6/ES2015, ES2020, ES2022, ESNext
        "target": "ES2020"
    }
}
```

不同目标的输出差异。

```ts
// target: ES5 - 使用 var
var greeting = "Hello";

// target: ES2020 - 使用 let/const
let greeting = "Hello";
```

### 模块系统

配置模块化方案。

```ts
{
    "compilerOptions": {
        "module": "commonjs",
        // 可选: none, commonjs, amd, umd, es6, es2020, esnext
    }
}
```

### 严格模式

strict选项启用所有严格类型检查，建议始终启用。

```ts
{
    "compilerOptions": {
        "strict": true,
        // 等同于开启以下所有选项：
        // "strictNullChecks": true,
        // "noImplicitAny": true,
        // "strictFunctionTypes": true,
        // 等等
    }
}
```

### 路径别名

配置路径别名简化导入。

```ts
{
    "compilerOptions": {
        "baseUrl": ".",
        //基础路径，"."指定到tsconfig.ts所在目录
        "paths": {
            "@/*": ["src/*"],
            //@映射到src/
            "@components/*": ["src/components/*"]
            //@components/映射到src/components/
        }
    }
}

import Button from "@components/Button";
// 从src/components/Button.ts导入Button
import { Header } from "@/components";
// 从src/components/Header.ts导入Header
```

### 包含与排除

控制哪些文件包含在编译中。

```ts
{
    "include": ["src/**/*"],
    "exclude": [
        "node_modules",
        "dist",
        "**/*.test.ts",
        "**/*.spec.ts"
    ]
}
```

## 编译选项

### 输出控制选项

```ts
{
    "compilerOptions": {
        // 输出目录
        "outDir": "./dist",

        // 源码根目录
        "rootDir": "./src",

        // 生成 .d.ts 声明文件
        "declaration": true,

        // 声明文件输出目录
        "declarationDir": "./types",

        // 生成 source map
        "sourceMap": true,

        // 生成 .js.map 文件
        "mapRoot": "./map"
    }
}
```

### 类型检查选项

```ts
{
    "compilerOptions": {
        // 严格模式（推荐始终开启）
        "strict": true,

        // 检查 null 和 undefined
        "strictNullChecks": true,

        // 检查 this 参数
        "noImplicitThis": true,

        // 严格函数类型
        "strictFunctionTypes": true,

        // 严格属性初始化
        "strictPropertyInitialization": true,

        // 不允许隐式 any
        "noImplicitAny": true,

        // 不允许返回 void
        "noImplicitReturns": true,

        // 开启所有严格检查
        "strict": true
    }
}
```

### 模块选项

```ts
{
    "compilerOptions": {
        // 模块系统
        "module": "commonjs",

        // 模块解析策略
        "moduleResolution": "node",

        // 解析基础路径
        "baseUrl": ".",

        // 路径别名
        "paths": {
            "@/*": ["src/*"]
        },

        // ES 模块互操作
        "esModuleInterop": true,

        // 允许默认导入
        "allowSyntheticDefaultImports": true,

        // 隔离模块
        "isolatedModules": true
    }
}
```

### ES特性选项

```ts
{
    "compilerOptions": {
        // 编译目标
        "target": "ES2020",

        // 启用的库
        "lib": ["ES2020", "DOM"],

        // 允许未使用的局部变量
        "noUnusedLocals": true,

        // 允许未使用的参数
        "noUnusedParameters": true,

        // 代码降级
        "downlevelIteration": true
    }
}
```

### 实验性选项

```ts
{
    "compilerOptions": {
        // 启用装饰器
        "experimentalDecorators": true,

        // 启用装饰器元数据
        "emitDecoratorMetadata": true,

        // 启用异步迭代器
        "emitDecoratorMetadata": true,

        // 跳过库检查
        "skipLibCheck": true
    }
}
```

### 常用编译组合

```ts
// Node.js 项目推荐配置
{
    "compilerOptions": {
        "target": "ES2020",//编译输出为ES2020标准的JavaScript
        "module": "commonjs",//使用CommonJS模块系统
        "lib": ["ES2020"],//编译时引入ES2020的类型定义
        "outDir": "./dist",//编译后的JS文件输出到./dist目录
        "rootDir": "./src",//源码目录为./src，编译器会保持次目录结构输出到outDir
        "strict": true,//开启所有严格类型检查
        "esModuleInterop": true,//允许CommonJS模块使用import x form 'y'语法
        "skipLibCheck": true,//跳过对声明文件的类型检查
        "forceConsistentCasingInFileNames": true,//强制文件名大小写一致
        "moduleResolution": "node",//使用Node.js的模块解析策略，自动查找node_modules
        "declaration": true//同时生成.d .ts类型声明文件，便于其他项目引用此包时获得类型提示
    }
}
```
